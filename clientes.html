// FUNÇÃO QUE BUSCA O PRODUTO NA INTERNET async function buscarProdutoNet() { const ean = document.getElementById('e_ean').value; const campoNome = document.getElementById('e_nome'); if (ean.length >= 8) { // Códigos de barra geralmente têm 8, 12 ou 13 dígitos campoNome.placeholder = "Buscando na nuvem..."; try { // Usando a API gratuita do OpenFoodFacts (funciona bem para bebidas e alimentos) const response = await fetch(`https://world.openfoodfacts.org/api/v0/product/${ean}.json`); const data = await response.json(); if (data.status === 1) { const produto = data.product; // Preenche o nome e tenta pegar a foto campoNome.value = produto.product_name ? produto.product_name.toUpperCase() : ""; if (produto.image_url) { document.getElementById('e_foto').value = produto.image_url; } alert("Produto localizado na rede!"); } else { campoNome.placeholder = "Não encontrado. Digite manualmente."; } } catch (error) { console.error("Erro na busca:", error); campoNome.placeholder = "Erro na conexão. Digite manualmente."; } } }
