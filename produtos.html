// MARGEM PADRÃO DO NEGÓCIO const MARGEM_PADRAO = 30; function dinheiro(v){ return v.toLocaleString('pt-BR',{style:'currency',currency:'BRL'}); } function arredondar(v){ return Math.ceil(v) - 0.01; } function calcularTudo(){ let custo = parseFloat(document.getElementById('custo-fardo').value) || 0; let qtd = parseFloat(document.getElementById('qtd-fardo').value) || 12; if(custo <= 0) return; // PREÇO DE VENDA BASEADO NA MARGEM let vendaFardo = custo / (1 - (MARGEM_PADRAO/100)); vendaFardo = arredondar(vendaFardo); let lucroFardo = vendaFardo - custo; let vendaUn = vendaFardo / qtd; // 💰 NOVO → CÁLCULO DO LUCRO POR UNIDADE let custoUn = custo / qtd; let lucroUn = vendaUn - custoUn; let markup = (lucroFardo / custo) * 100; // ATUALIZA TELA document.getElementById('venda-fardo').value = dinheiro(vendaFardo); document.getElementById('venda-un').value = dinheiro(vendaUn); document.getElementById('lucro-dinheiro').innerText = dinheiro(lucroFardo); document.getElementById('lucro-unidade').innerText = dinheiro(lucroUn); document.getElementById('margem-final').innerText = MARGEM_PADRAO + "%"; document.getElementById('markup-real').innerText = markup.toFixed(1) + "%"; } // SALVAR PRODUTO NO BANCO LOCAL function salvarProduto(){ let produto = { nome: document.getElementById('nome-cerv').value, codigo: document.getElementById('cod-barras').value, custoFardo: document.getElementById('custo-fardo').value, vendaFardo: document.getElementById('venda-fardo').value, vendaUn: document.getElementById('venda-un').value }; let banco = JSON.parse(localStorage.getItem("produtos") || "[]"); banco.push(produto); localStorage.setItem("produtos", JSON.stringify(banco)); alert("Produto salvo com sucesso!"); }
