// LÓGICA 2: DESTILADOS (Com Margem de Perda Realista) function calcularD() { const custoGarrafa = parseFloat(document.getElementById('d_custo_garrafa').value) || 0; const mlGarrafa = parseFloat(document.getElementById('d_ml_garrafa').value) || 1; const mlDose = parseFloat(document.getElementById('d_ml_dose').value) || 1; const markup = parseFloat(document.getElementById('d_markup').value) || 0; // Fator de perda: Geralmente 10% para destilados (derramamento/fundo de garrafa) // Isso faz com que 1000ml rendam aprox. 18 doses de 50ml em vez de 20. const margemPerda = 0.10; const mlUtil = mlGarrafa * (1 - margemPerda); const totalDosesReais = mlUtil / mlDose; const custoPorDoseReal = custoGarrafa / totalDosesReais; const precoVendaDose = custoPorDoseReal * (1 + (markup / 100)); // Atualiza a tela com os valores reais document.getElementById('d_venda').innerText = `R$ ${precoVendaDose.toFixed(2)}`; document.getElementById('d_detalhe').innerText = `Rende aprox. ${Math.floor(totalDosesReais)} doses (c/ perda) | Custo Real/Dose: R$ ${custoPorDoseReal.toFixed(2)}`; }
